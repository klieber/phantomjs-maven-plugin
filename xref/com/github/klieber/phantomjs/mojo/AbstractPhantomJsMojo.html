21  package com.github.klieber.phantomjs.mojo;
22  
23  import org.apache.maven.plugin.AbstractMojo;
24  import org.apache.maven.plugin.MojoExecutionException;
25  import org.apache.maven.plugins.annotations.Parameter;
26  import org.apache.maven.project.MavenProject;
27  import org.codehaus.plexus.util.StringUtils;
28  
29  /**
30   * Abstract base class for phantomjs-maven-plugin mojos.
31   */
32  public abstract class AbstractPhantomJsMojo extends AbstractMojo {
33  
34    /**
35     * The name of the property that will contains the path to the binary.
36     *
37     * @since 0.1
38     */
39    @Parameter(
40        defaultValue = "phantomjs.binary",
41        property = "phantomjs.propertyName",
42        required = true
43    )
44    private String propertyName;
45  
46    /**
47     * The path to the phantomjs binary
48     *
49     * @since 0.2
50     */
51    @Parameter(
52        property = "phantomjs.binary"
53    )
54    private String phantomJsBinary;
55  
56    /**
57     * Skip the phantomjs-maven-plugin execution.
58     *
59     * @since 0.2
60     */
61    @Parameter(
62        defaultValue = "false",
63        property = "phantomjs.skip",
64        required = true
65    )
66    private boolean skip;
67  
68    @Parameter(defaultValue = "${project}", readonly = true)
69    private MavenProject mavenProject;
70  
71    public final void execute() throws MojoExecutionException {
72      if (!skip) {
73        this.run();
74      }
75    }
76  
77    protected abstract void run() throws MojoExecutionException;
78  
79    protected String getPhantomJsBinary() {
80      if (StringUtils.isBlank(this.phantomJsBinary)) {
81        this.phantomJsBinary = mavenProject.getProperties().getProperty(this.propertyName);
82      }
83      return this.phantomJsBinary;
84    }
85  
86    protected void setPhantomJsBinary(String binary) {
87      mavenProject.getProperties().setProperty(this.propertyName, binary);
88    }
89  }
