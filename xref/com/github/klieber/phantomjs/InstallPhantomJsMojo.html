21  package com.github.klieber.phantomjs;
22  
23  import java.io.File;
24  import java.io.FileOutputStream;
25  import java.io.IOException;
26  import java.net.MalformedURLException;
27  import java.net.URL;
28  import java.nio.channels.Channels;
29  import java.nio.channels.ReadableByteChannel;
30  
31  import org.apache.maven.plugin.AbstractMojo;
32  import org.apache.maven.plugin.MojoExecutionException;
33  import org.apache.maven.plugins.annotations.LifecyclePhase;
34  import org.apache.maven.plugins.annotations.Mojo;
35  import org.apache.maven.plugins.annotations.Parameter;
36  import org.apache.maven.project.MavenProject;
37  
38  import de.schlichtherle.truezip.file.TFile;
39  
40  /**
41   * Maven plugin for downloading and installing phantomjs binaries.
42   */
43  @Mojo( name = "install", defaultPhase = LifecyclePhase.PROCESS_TEST_SOURCES )
44  public class InstallPhantomJsMojo extends AbstractMojo {
45  
46    /**
47     * The version of phantomjs to install.
48     */
49    @Parameter(
50        property = "phantomjs.version",
51        required=true
52    )
53    private String version;
54  
55    /**
56     * The base url the phantomjs binary can be downloaded from.
57     */
58    @Parameter(
59        defaultValue="https://phantomjs.googlecode.com/files/",
60        property = "phantomjs.baseUrl",
61        required=true
62    )
63    private String baseUrl;
64  
65    /**
66     * The directory the phantomjs binary should be installed.
67     */
68    @Parameter(
69        defaultValue = "${project.build.directory}/phantomjs-maven-plugin",
70        property = "phantomjs.outputDir",
71        required = true
72    )
73    private File outputDirectory;
74  
75    /**
76     * The name of the property that will contains the path to the binary.
77     */
78    @Parameter(
79        defaultValue = "phantomjs.binary",
80        property = "phantomjs.propertyName",
81        required = true
82    )
83    private String propertyName;
84  
85    @Parameter(defaultValue="${project}",readonly=true)
86    private MavenProject mavenProject;
87  
88    public void execute() throws MojoExecutionException {
89  
90      if (!outputDirectory.exists() && !outputDirectory.mkdirs()) {
91        throw new MojoExecutionException("unable to create directory: " + outputDirectory);
92      }
93  
94      PhantomJSArchive phantomJSFile = new PhantomJSArchiveBuilder(version).build();
95  
96      File extractTo = new File(outputDirectory,phantomJSFile.getExtractToPath());
97      if (!extractTo.exists()) {
98        StringBuilder url = new StringBuilder();
99        url.append(baseUrl);
100       url.append(phantomJSFile.getArchiveName());
101 
102       try {
103         URL downloadLocation = new URL(url.toString());
104 
105         getLog().info("Downloading phantomjs binaries from " + url);
106         ReadableByteChannel rbc = Channels.newChannel(downloadLocation.openStream());
107         File outputFile = new File(outputDirectory, phantomJSFile.getArchiveName());
108         FileOutputStream fos = new FileOutputStream(outputFile);
109         fos.getChannel().transferFrom(rbc, 0, Long.MAX_VALUE);
110         TFile archive = new TFile(outputDirectory, phantomJSFile.getPathToExecutable());
111 
112         getLog().info("Extracting "+archive.getAbsolutePath()+" to " + extractTo.getAbsolutePath());
113         extractTo.getParentFile().mkdirs();
114         archive.cp(extractTo);
115         extractTo.setExecutable(true);
116       } catch (MalformedURLException e) {
117         throw new MojoExecutionException("Unable to download phantomjs binary from " + url, e);
118       } catch (IOException e) {
119         throw new MojoExecutionException("Unable to download phantomjs binary from " + url, e);
120       }
121     }
122     mavenProject.getProperties().put(propertyName, extractTo.getAbsolutePath());
123   }
124 }
