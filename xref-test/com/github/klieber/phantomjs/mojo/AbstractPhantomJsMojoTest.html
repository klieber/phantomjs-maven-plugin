21  package com.github.klieber.phantomjs.mojo;
22  
23  import org.apache.maven.plugin.MojoExecutionException;
24  import org.apache.maven.project.MavenProject;
25  import org.junit.Test;
26  import org.junit.runner.RunWith;
27  import org.mockito.Mock;
28  import org.mockito.Spy;
29  import org.mockito.internal.util.reflection.Whitebox;
30  import org.mockito.runners.MockitoJUnitRunner;
31  
32  import java.util.Properties;
33  
34  import static org.junit.Assert.assertEquals;
35  import static org.mockito.Mockito.*;
36  
37  @RunWith(MockitoJUnitRunner.class)
38  public class AbstractPhantomJsMojoTest {
39  
40    private static final String BINARY_PATH = "bin/phantomjs";
41    private static final String PROPERTY_NAME = "phantomjs.bin.path";
42  
43    @Mock
44    private MavenProject project;
45  
46    @Mock
47    private Properties properties;
48  
49    @Spy
50    private AbstractPhantomJsMojo mojo = new AbstractPhantomJsMojo() {
51      @Override
52      protected void run() throws MojoExecutionException { }
53    };
54  
55    @Test
56    public void testExecute() throws MojoExecutionException {
57      mojo.execute();
58      verify(mojo).run();
59    }
60  
61    @Test
62    public void testSkipExecute() throws MojoExecutionException {
63      Whitebox.setInternalState(mojo,"skip",true);
64      mojo.execute();
65      verifyNoMoreInteractions(mojo);
66    }
67  
68    @Test
69    public void testGetPhantomJsBinary() {
70      Whitebox.setInternalState(mojo,"phantomJsBinary", BINARY_PATH);
71      assertEquals(BINARY_PATH, mojo.getPhantomJsBinary());
72    }
73  
74    @Test
75    public void testGetPhantomJsBinaryFromProject() {
76      when(project.getProperties()).thenReturn(properties);
77      when(properties.getProperty(PROPERTY_NAME)).thenReturn(BINARY_PATH);
78      Whitebox.setInternalState(mojo,"mavenProject",project);
79      Whitebox.setInternalState(mojo,"propertyName", PROPERTY_NAME);
80      assertEquals(BINARY_PATH, mojo.getPhantomJsBinary());
81      verify(project,times(1)).getProperties();
82      verify(properties,times(1)).getProperty(PROPERTY_NAME);
83    }
84  
85    @Test
86    public void testSetPhantomJsBinary() {
87      when(project.getProperties()).thenReturn(properties);
88      Whitebox.setInternalState(mojo,"mavenProject",project);
89      Whitebox.setInternalState(mojo, "propertyName", PROPERTY_NAME);
90      mojo.setPhantomJsBinary(BINARY_PATH);
91      verify(properties).setProperty(PROPERTY_NAME, BINARY_PATH);
92    }
93  }
